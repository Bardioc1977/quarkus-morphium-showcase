Entity Definition

@Entity(collectionName = "products")
@Index({"-price, name"})
@Cache(maxEntries = 100, strategy = LRU, timeout = 30000)
@WriteSafety(level = NORMAL)
@DefaultReadPreference(PRIMARY)
public class Product {
    @Id
    private MorphiumId id;
    @Index
    private String name;
    @Property(fieldName = "product_description")
    private String description;
    @Index
    private double price;
    private Category category; // @Embedded
}

Service Code (Morphium API)

// Store a product
morphium.store(product);

// Search by name (regex)
morphium.createQueryFor(Product.class)
    .f(Product.Fields.name).matches("(?i)laptop")
    .asList();

// Price range query with sort
morphium.createQueryFor(Product.class)
    .f(Product.Fields.price).gte(min)
    .f(Product.Fields.price).lte(max)
    .sort(Map.of(Product.Fields.price, 1))
    .asList();

// Batch insert
morphium.storeList(products);

// Delete + Cache clear
morphium.delete(product);
morphium.getCache().clearCachefor(Product.class);

// Ensure indexes on startup
morphium.ensureIndicesFor(Product.class);
