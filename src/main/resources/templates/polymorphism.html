Polymorphism Code

@Entity(collectionName = "vehicles", polymorph = true)
public class Vehicle {
    @Id private MorphiumId id;
    private String manufacturer, model;
}

@Entity(collectionName = "vehicles", polymorph = true)
public class Car extends Vehicle {
    private int doors;
    private String fuelType;
}

// Query Vehicle.class returns BOTH Car and Truck instances!
List<Vehicle> all = morphium.createQueryFor(Vehicle.class).asList();
all.forEach(v -> {
    if (v instanceof Car car) { /* car-specific */ }
    if (v instanceof Truck truck) { /* truck-specific */ }
});

Service Code (Morphium API)

// Polymorphic query: Vehicle.class returns Car + Truck
List<Vehicle> all = morphium.createQueryFor(Vehicle.class)
    .sort(Map.of(Vehicle.Fields.manufacturer, 1))
    .asList();

// Store a Car - Morphium auto-stores class_name discriminator
Car car = new Car();
car.setManufacturer("Porsche");
morphium.store(car);

// Drop entire polymorphic collection
morphium.dropCollection(Vehicle.class);

// Batch store mixed types
morphium.storeList(List.of(car1, car2, truck1));

Custom Type Mapper (MorphiumTypeMapper<Money>)

public class MoneyMapper implements MorphiumTypeMapper<Money> {
    public Object marshall(Money o) {
        return Map.of("amount", o.getAmount(), "currency", o.getCurrency());
    }
    public Money unmarshall(Object d) {
        Map<?,?> map = (Map<?,?>) d;
        return new Money(
            ((Number) map.get("amount")).doubleValue(),
            (String) map.get("currency")
        );
    }
}
