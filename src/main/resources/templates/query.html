Entity Definition

@Entity(collectionName = "employees")
@Index({"lastName, firstName"})
@Lifecycle
public class Employee {
    @Id private MorphiumId id;
    @Aliases({"mail", "e_mail"}) private String email;
    @Transient private String fullName; // computed, not stored
    @IgnoreNullFromDB private List<String> skills;
    @PostLoad void onLoad() { fullName = firstName + " " + lastName; }
}

Service Code (Morphium API)

// Equality filter
query.f(Employee.Fields.department).eq("Engineering");

// Range: gte + lte
query.f(Employee.Fields.salary).gte(50000);
query.f(Employee.Fields.salary).lte(100000);

// Regex matching
query.f(Employee.Fields.lastName).matches("^S.*");

// In / Not In
query.f(Employee.Fields.skills).in(List.of("Java", "Python"));
query.f(Employee.Fields.department).nin(List.of("HR"));

// Exists check
query.f(Employee.Fields.email).exists();

// Pagination: sort + skip + limit
query.sort(Map.of("salary", -1)).skip(20).limit(10);

// Distinct values
query.distinct("department");
