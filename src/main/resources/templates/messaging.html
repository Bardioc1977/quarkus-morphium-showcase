// Atomic update: mark message as read
var query = morphium.createQueryFor(ChatMessage.class)
    .f(ChatMessage.Fields.id).eq(new MorphiumId(id));
query.set(ChatMessage.Fields.read, true, false, false, null);

// Topic-based filtering with sort
morphium.createQueryFor(ChatMessage.class)
    .f(ChatMessage.Fields.topic).eq("general")
    .sort(Map.of(ChatMessage.Fields.sentAt, -1))
    .asList();

// Count all messages
morphium.createQueryFor(ChatMessage.class).countAll();

Morphium Built-in Messaging System

Morphium includes a built-in messaging/queuing system that uses MongoDB as the transport layer. It supports topics, exclusive messages, and broadcast patterns without any external message broker. 

// Create a messaging instance
Messaging messaging = morphium.createMessaging();

// Listen for messages on a topic
messaging.addListenerForTopic("orders", (msg) -> {
    System.out.println("Received: " + msg.getValue());
    return null;
});

// Send a message to a topic
Msg msg = new Msg("orders", "New order received", "...");
messaging.sendMessage(msg);
