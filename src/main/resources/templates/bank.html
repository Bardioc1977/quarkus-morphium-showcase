Transaction Code

@MorphiumTransactional
public TransferResult transfer(String from, String to, double amount) {
    Account src = findAccountByNumber(from);
    Account dst = findAccountByNumber(to);
    if (src.getBalance() < amount) throw new InsufficientFundsException();

    // Atomic balance changes
    morphium.inc(queryFor(src), "balance", -amount);
    morphium.inc(queryFor(dst), "balance", amount);

    Transfer transfer = new Transfer(from, to, amount);
    morphium.store(transfer);
    return new TransferResult(true, "Transfer complete", transfer);
}

Service Code (Morphium API)

// Transactional transfer
@MorphiumTransactional
public TransferResult transfer(String from, String to, double amount) {
    // Atomic decrement on source
    var fromQuery = morphium.createQueryFor(Account.class)
        .f(Account.Fields.accountNumber).eq(from);
    morphium.inc(fromQuery, Account.Fields.balance, -amount);

    // Atomic increment on target
    var toQuery = morphium.createQueryFor(Account.class)
        .f(Account.Fields.accountNumber).eq(to);
    morphium.inc(toQuery, Account.Fields.balance, amount);

    morphium.store(transfer); // record the transfer
}
